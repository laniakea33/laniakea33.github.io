//	클래스 전체를 제네릭으로 만듦
class AAA<T> {
	T x;
	T y;
	AAA(T x, T y) {
		this.x = x;
		this.y = y;
	}
	
	public void showTest() {
		System.out.print(x + ", ");
		System.out.println(y);
	}
}

//	특정 함수만 제네릭으로 만듦
class Box {	
	//	이 형식은 꼭 지켜야 함
	public <T1, T2> void show(T1 x, T2 x2) {
		System.out.print(x + ", ");
		System.out.println(x2);
	}
}
